<script>
const COLS = 24, ROWS = 13;
const NOTE_NAMES = ['도','레','미','파','솔','라','시','도²','레²','미²','파²','솔²','라²'];
const NOTE_FREQS = [523.25,587.33,659.25,698.46,783.99,880.00,987.77,1046.50,1174.66,1318.51,1396.91,1567.98,1760.00];

const cv = document.getElementById('cv');
const ctx = cv.getContext('2d');
const DPR = window.devicePixelRatio || 1;
const W = 568, H = 380;
cv.width = W * DPR; cv.height = H * DPR;
cv.style.width = W + 'px'; cv.style.height = H + 'px';
ctx.scale(DPR, DPR);

const PAD_L = 36, PAD_T = 12, PAD_B = 12;
const gridW = W - PAD_L - 8, gridH = H - PAD_T - PAD_B;
const cellW = gridW / COLS, cellH = gridH / ROWS;

let grid = Array.from({length: ROWS}, () => new Array(COLS).fill(false));
let playing = false, playhead = -1, tempo = 120, lastStepTime = 0, audioCtx = null;

function ensureAudio() {
  if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
  if (audioCtx.state === 'suspended') audioCtx.resume();
}

function playNote(freq, t) {
  if (!audioCtx) return;
  const osc = audioCtx.createOscillator(), gain = audioCtx.createGain();
  osc.type = 'sine'; osc.frequency.value = freq;
  const partial = audioCtx.createOscillator(), pgain = audioCtx.createGain();
  partial.type = 'triangle'; partial.frequency.value = freq * 2; pgain.gain.value = 0.1;
  partial.connect(pgain); pgain.connect(audioCtx.destination);
  gain.gain.setValueAtTime(0.0001, t);
  gain.gain.exponentialRampToValueAtTime(0.26, t + 0.01);
  gain.gain.exponentialRampToValueAtTime(0.0001, t + 0.7);
  osc.connect(gain); gain.connect(audioCtx.destination);
  osc.start(t); osc.stop(t + 0.7);
  partial.start(t); partial.stop(t + 0.35);
}

function activeCount() {
  let c = 0;
  for (let r = 0; r < ROWS; r++) for (let col = 0; col < COLS; col++) if (grid[r][col]) c++;
  return c;
}

function isDark() { return window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches; }
function gridLine() { return isDark() ? 'rgba(255,255,255,0.08)' : 'rgba(0,0,0,0.08)'; }
function labelColor() { return isDark() ? '#888' : '#999'; }

function draw() {
  ctx.clearRect(0, 0, W, H);
  ctx.font = '10px Pretendard, sans-serif';
  ctx.textAlign = 'right'; ctx.textBaseline = 'middle';
  for (let r = 0; r < ROWS; r++) {
    const cy = PAD_T + r * cellH + cellH/2;
    ctx.fillStyle = labelColor();
    ctx.fillText(NOTE_NAMES[ROWS - 1 - r], PAD_L - 6, cy);
  }
  for (let col = 0; col < COLS; col++) {
    for (let r = 0; r < ROWS; r++) {
      const x = PAD_L + col * cellW, y = PAD_T + r * cellH;
      if (col === playhead) { ctx.fillStyle = 'rgba(127,119,221,0.12)'; ctx.fillRect(x, PAD_T, cellW, gridH); }
      ctx.strokeStyle = gridLine(); ctx.lineWidth = 0.5;
      ctx.strokeRect(x, y, cellW, cellH);
    }
  }
  for (let r = 0; r < ROWS; r++) {
    for (let col = 0; col < COLS; col++) {
      if (!grid[r][col]) continue;
      const cx = PAD_L + col * cellW + cellW/2, cy = PAD_T + r * cellH + cellH/2;
      const isHit = col === playhead && playing;
      const radius = isHit ? 8 : 5.5;
      ctx.beginPath(); ctx.arc(cx, cy, radius, 0, Math.PI*2);
      ctx.fillStyle = isHit ? '#7F77DD' : '#AFA9EC'; ctx.fill();
      if (isHit) {
        ctx.beginPath(); ctx.arc(cx, cy, radius + 4, 0, Math.PI*2);
        ctx.strokeStyle = 'rgba(127,119,221,0.4)'; ctx.lineWidth = 2; ctx.stroke();
      }
    }
  }
  if (playing && playhead >= 0) {
    const px = PAD_L + playhead * cellW + cellW/2;
    ctx.beginPath(); ctx.moveTo(px, PAD_T); ctx.lineTo(px, PAD_T + gridH);
    ctx.strokeStyle = '#7F77DD'; ctx.lineWidth = 1.5; ctx.stroke();
  }
}

cv.addEventListener('click', (e) => {
  const rect = cv.getBoundingClientRect();
  const x = e.clientX - rect.left - PAD_L, y = e.clientY - rect.top - PAD_T;
  if (x < 0 || y < 0 || x > gridW || y > gridH) return;
  const col = Math.floor(x / cellW), r = Math.floor(y / cellH);
  if (col < 0 || col >= COLS || r < 0 || r >= ROWS) return;
  grid[r][col] = !grid[r][col];
  ensureAudio();
  if (grid[r][col]) playNote(NOTE_FREQS[ROWS - 1 - r], audioCtx.currentTime);
  clearActivePreset();
  draw(); updateProps();
});

function stepInterval() { return (60 / tempo) * 1000 / 2; }

function tick(now) {
  if (!playing) return;
  if (now - lastStepTime >= stepInterval()) {
    lastStepTime = now;
    playhead = (playhead + 1) % COLS;
    ensureAudio();
    for (let r = 0; r < ROWS; r++) if (grid[r][playhead]) playNote(NOTE_FREQS[ROWS - 1 - r], audioCtx.currentTime);
    updateProps();
  }
  draw();
  requestAnimationFrame(tick);
}

function togglePlay() {
  ensureAudio();
  playing = !playing;
  const btn = document.getElementById('btnPlay'), ico = document.getElementById('ico');
  const lbl = document.getElementById('blabel'), badge = document.getElementById('badge');
  if (playing) {
    playhead = -1; lastStepTime = 0;
    btn.classList.add('run');
    ico.innerHTML = '<rect x="6" y="4" width="4" height="16"/><rect x="14" y="4" width="4" height="16"/>';
    lbl.textContent = '정지';
    badge.className = 'mb-badge run';
    document.getElementById('badge-t').textContent = 'RUNNING';
    requestAnimationFrame(tick);
  } else {
    playhead = -1;
    btn.classList.remove('run');
    ico.innerHTML = '<polygon points="5 3 19 12 5 21 5 3"/>';
    lbl.textContent = '재생';
    badge.className = 'mb-badge';
    document.getElementById('badge-t').textContent = 'IDLE';
    draw();
  }
  updateProps();
}

function clearGrid() { grid = Array.from({length: ROWS}, () => new Array(COLS).fill(false)); clearActivePreset(); draw(); updateProps(); }
function setTempo(v) { tempo = +v; document.getElementById('tempo-v').textContent = v + ' BPM'; }
function clearActivePreset() { document.querySelectorAll('.mb-chip').forEach(c => c.classList.remove('active')); }

function setNotes(seq) {
  grid = Array.from({length: ROWS}, () => new Array(COLS).fill(false));
  seq.forEach((note, col) => { if (note >= 0 && note < ROWS && col < COLS) grid[ROWS-1-note][col] = true; });
}

function loadPreset(name, el) {
  if (name === 'twinkle') {
    setNotes([0,0,4,4,5,5,4,-1,3,3,2,2,1,1,0,-1,4,4,3,3,2,2,1,-1]);
  } else if (name === 'elise') {
    setNotes([7,6,7,6,7,2,5,3,0,-1,-1,0,4,7,9,-1,-1,4,6,9,11,-1,-1,7]);
  } else if (name === 'canon') {
    setNotes([7,5,4,2,0,-1,2,4,-1,4,2,4,5,4,2,0,-1,2,0,2,4,5,7,9]);
  } else if (name === 'waltz') {
    setNotes([0,-1,4,7,-1,9,11,-1,7,4,-1,2,0,-1,4,7,-1,11,9,-1,7,4,-1,0]);
  } else {
    const seq = [];
    for (let col = 0; col < COLS; col++) seq.push(Math.random() > 0.5 ? Math.floor(Math.random()*ROWS) : -1);
    setNotes(seq);
  }
  clearActivePreset();
  if (el) el.classList.add('active');
  draw(); updateProps();
}

function updateProps() {
  const total = activeCount();
  const angle = playing && playhead >= 0 ? Math.round((playhead / COLS) * 360) : 0;
  const progress = playing && playhead >= 0 ? Math.round((playhead / (COLS-1)) * 100) : 0;
  document.getElementById('p-trigger').textContent = playing ? '1' : '0';
  document.getElementById('p-resource').textContent = total + ' pins';
  document.getElementById('p-angle').textContent = angle + '°';
  document.getElementById('p-progress').textContent = progress + '%';
}

loadPreset('waltz', document.querySelectorAll('.mb-chip')[3]);

if (window.matchMedia) {
  window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', draw);
}
</script>
